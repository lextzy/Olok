SELECT C.CUS_CODE, C.CUS_BALANCE
FROM CUSTOMER C, INVOICE I
WHERE C.CUS_CODE NOT IN (SELECT CUS_CODE FROM INVOICE)
GROUP BY C.CUS_CODE
ORDER BY C.CUS_CODE;

WHERE C.CUS_CODE NOT IN (SELECT CUS_CODE FROM INVOICE);
